### algoritma_menghitung_volume_bola
deklarasi\
	r, volume: double\
deskripsi\
    read(r)\
    volume <- 4/3.0 * 3.14 * (r * r * r)\
    write(volume)\
			  
### algoritma_menghitung_luas_segitiga
deklarasi\
    a, t, luas: double\
deskripsi\
    read(a, t)\
    luas <- 1/2 (a * t)\
    write(luas)\
  
### algoritma_operasi_matematika
deklarasi\
     x, y, kurang, tambah, kali, bagi: int\
  deskripsi\
     read(x, y)\
     kurang <- x-y\
     tambah <- x+y\
     kali <- x*y\
     bagi <- x/y\
     write(kurang)\
     write(tambah\
     write(kali)\
     write(bagi)
