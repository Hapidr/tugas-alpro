Repository ini untuk upload file tugas alpro

### algoritma_menghitung_volume_bola
deklarasi\
	r, volume: double\
deskripsi\
    read(r)\
    volume <- 4/3.0 * 3.14 * (r * r * r)\
    write(volume)\
			  
### algoritma_menghitung_luas_segitiga
deklarasi\
    a, t, luas: double\
deskripsi\
    read(a, t)\
    luas <- 1/2 (a * t)\
    write(luas)\
  
### algoritma_operasi_matematika
deklarasi\
   |  x, y: int\
  deskripsi\
   |  read(x, y)\
   |  write(x - y)\
   |  write(x + y)\
   |  write(x * y)\
   |  write(x / y)
